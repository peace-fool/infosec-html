Windows Privilege Escalation

Basic System Enumeration

System/User/Group Enumeration

# Basics
systeminfo
hostname

# Who am I?
whoami
echo %username%

# User/Group/Privilages info
whoami /all

# What users/localgroups are on the machine?
net users
net localgroups

# More info about a specific user. Check if user has privileges.
net user user1

# View Domain Groups
net group /domain

# View Members of Domain Group
net group /domain <Group Name>

# How well patched is the system?
wmic qfe get Caption,Description,HotFixID,InstalledOn

# Listing Connected Drives
wmic logicaldisk get caption,description,providername

Network Enumeration

# Network
ipconfig /all
route print

# Listing all Internal and External Network 
# with thier Corresponding IP and Port
netstat -ano

# (ARP) Address Resolution Protocol
# ARP is a protocol or procedure that connects an ever-changing IP -
# address to a MAC address, in a local-area network (LAN).
arp -A

Firewall/Antivirus Enumeration

# Firewall
netsh firewall show state
netsh firewall show config

# AV
sc query windefend
sc queryex type= service

Clear Text Passwords

Generig Searching

findstr /si password *.txt
findstr /si password *.xml
findstr /si password *.ini

#Find all those strings in config files.
dir /s *pass* == *cred* == *vnc* == *.config*

# Find all passwords in all files.
findstr /spin "password" *.*
findstr /spin "password" *.*

Searching Registry

# VNC
reg query "HKCU\Software\ORL\WinVNC3\Password"

# Windows autologin
reg query "HKLM\SOFTWARE\Microsoft\Windows NT\Currentversion\Winlogon"

# SNMP Paramters
reg query "HKLM\SYSTEM\Current\ControlSet\Services\SNMP"

# Putty
reg query "HKCU\Software\SimonTatham\PuTTY\Sessions"

# Search for password in registry
reg query HKLM /f password /t REG_SZ /s
reg query HKCU /f password /t REG_SZ /s
